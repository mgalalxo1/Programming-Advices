#include <iostream>
using namespace std;

int main() {
    float mark1, mark2, mark3;

    cout << "Please enter the first mark: ";
    cin >> mark1;

    cout << "Please enter the second mark: ";
    cin >> mark2;

    cout << "Please enter the third mark: ";
    cin >> mark3;

    float average = (mark1 + mark2 + mark3) / 3;

    cout << "The average of the marks is: " << average << endl;

    return 0;
}
