#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    while (n <= 0) {
        cout << "Factorial Must be Positive Number" << endl;
        cin >> n;
    }
    int factorial = 1;
    for (int i = n; i >= 1; i--)
        factorial *= i;
    cout << factorial;
    return 0;
}
