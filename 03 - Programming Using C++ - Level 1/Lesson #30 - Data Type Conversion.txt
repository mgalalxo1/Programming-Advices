#include <iostream>
#include <string>
using namespace std;

int main() {
    string st1 = "43.22";
    double d = stod(st1);
    float f = stof(st1);
    int i = static_cast<int>(d); 

    cout << "Original string: " << st1 << endl;
    cout << "As double: " << d << endl;
    cout << "As float: " << f << endl;
    cout << "As int: " << i << endl;

    cout << "--------------------------" << endl;

    int N1 = 20;
    string s1 = to_string(N1);
    cout << "Integer N1: " << N1 << " as string: " << s1 << endl;

    double N2 = 33.5;
    string s2 = to_string(N2);
    cout << "Double N2: " << N2 << " as string: " << s2 << endl;

    float N3 = 55.23f;
    string s3 = to_string(N3);
    int i3 = static_cast<int>(N3);
    cout << "Float N3: " << N3 << " as string: " << s3 << ", as int: " << i3 << endl;

    return 0;
}
