#include <iostream>
using namespace std;

int main() {
    float loanAmount;
    int months;

    cout << "Enter the loan amount: ";
    cin >> loanAmount;

    cout << "Enter how many months you want to pay it over: ";
    cin >> months;

    float monthlyInstallment = loanAmount / months;

    cout << "Your monthly installment is: " << monthlyInstallment << endl;

    return 0;
}
