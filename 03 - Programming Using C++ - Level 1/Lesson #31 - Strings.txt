#include <iostream>
#include <string>
using namespace std;

int main() {
    string str1, str2, str3;

    cout << "Please enter String1: ";
    getline(cin, str1);

    cout << "Please enter String2: ";
    cin >> str2;

    cout << "Please enter String3: ";
    cin >> str3;

    cout << "\n===============================" << endl;
    cout << "The Length of String1 is: " << str1.length() << endl;

    cout << "Characters at 0,2,4,7 are: ";
    if (str1.length() > 7) {
        cout << str1[0] << ", " << str1[2] << ", " << str1[4] << ", " << str1[7] << endl;
    } else {
        cout << "String1 is too short to access all indices." << endl;
    }

    string concat = str2 + str3;
    cout << "Concatenating String2 and String3 = " << concat << endl;

    int num2 = stoi(str2);
    int num3 = stoi(str3);
    cout << str2 << " + " << str3 << " = " << num2 + num3 << endl;

    return 0;
}
