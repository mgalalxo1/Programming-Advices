#include <iostream>
using namespace std;

int main() {
    float loanAmount, monthlyInstallment;

    cout << "Enter the loan amount: ";
    cin >> loanAmount;

    cout << "Enter the monthly installment: ";
    cin >> monthlyInstallment;

    int totalMonths = loanAmount / monthlyInstallment;

    cout << "You need " << totalMonths << " months to pay off the loan." << endl;

    return 0;
}
