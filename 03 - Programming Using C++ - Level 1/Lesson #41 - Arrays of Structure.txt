#include <iostream>
using namespace std;

struct Person {
    string firstName;
    string lastName;
    int age;
    string phone;
};

void printPerson(const Person& p) {
    cout << "********************\n";
    cout << "FirstName: " << p.firstName << endl;
    cout << "LastName : " << p.lastName << endl;
    cout << "Age      : " << p.age << " Years" << endl;
    cout << "Phone    : " << p.phone << endl;
    cout << "********************\n";
}

int main() {
    Person people[2] = {
        {"Mohammed", "Abu-Hadhoud", 44, "8982788"},
        {"Ali", "Fadi", 22, "5234222"}
    };

    for (int i = 0; i < 2; i++) {
        printPerson(people[i]);
    }

    return 0;
}
