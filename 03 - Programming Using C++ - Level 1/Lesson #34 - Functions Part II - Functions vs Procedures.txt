#include <iostream>
using namespace std;


int addFunction(int a, int b) {
    return a + b;
}


void printResult(int result) {
    cout << "======================" << endl;
    cout << result << endl;
}

int main() {
    int num1, num2;

    cout << "Please enter Number1: ";
    cin >> num1;

    cout << "Please enter Number2: ";
    cin >> num2;

    int result = addFunction(num1, num2); 
    printResult(result);                  
    return 0;
}
