#include <iostream>
#include <cmath>
using namespace std;

void PrintDescribedCircleArea(double a, double b, double c) {
    double PI = 3.14;
    double p = (a + b + c) / 2;
    double T = (a * b * c) / (4 * sqrt(p * (p - a) * (p - b) * (p - c)));
    double area = PI * T * T;
    cout << area;
}

int main() {
    double a, b, c;
    cin >> a >> b >> c;
    PrintDescribedCircleArea(a, b, c);
    return 0;
}
