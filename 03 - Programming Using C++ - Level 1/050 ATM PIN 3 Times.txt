#include <iostream>
using namespace std;

int main() {
    int pin, counter = 0;
    const int correctPin = 1234;
    const int balance = 7500;

    while (counter < 3) {
        cin >> pin;

        if (pin == correctPin) {
            cout << "Your Balance is: " << balance;
            return 0;
        } else {
            cout << "Wrong PIN" << endl;
            counter++;
        }
    }

    cout << "Card is locked!";
    return 0;
}
