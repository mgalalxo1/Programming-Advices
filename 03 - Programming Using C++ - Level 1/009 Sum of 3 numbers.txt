#include <iostream>
using namespace std;

int main() {
    int num1, num2, num3;

    cout << "Please enter the first number: ";
    cin >> num1;

    cout << "Please enter the second number: ";
    cin >> num2;

    cout << "Please enter the third number: ";
    cin >> num3;

    int sum = num1 + num2 + num3;

    cout << "The sum of the three numbers is: " << sum << endl;

    return 0;
}
