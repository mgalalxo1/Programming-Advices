#include <iostream>
using namespace std;

void SwapAndPrint(int num1, int num2) {
    cout << num1 << endl;
    cout << num2 << endl;

    int temp = num1;
    num1 = num2;
    num2 = temp;

    cout << num1 << endl;
    cout << num2 << endl;
}

int main() {
    int a, b;
    cin >> a >> b;
    SwapAndPrint(a, b);
    return 0;
}
