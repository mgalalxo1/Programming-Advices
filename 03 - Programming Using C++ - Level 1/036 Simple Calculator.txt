#include <iostream>
using namespace std;

int main() {
    double num1, num2;
    char op;
    cin >> num1 >> num2 >> op;

    if (op == '+')
        cout << num1 + num2;
    else if (op == '-')
        cout << num1 - num2;
    else if (op == '*')
        cout << num1 * num2;
    else if (op == '/')
        cout << num1 / num2;
    else
        cout << "Wrong Operation";

    return 0;
}
