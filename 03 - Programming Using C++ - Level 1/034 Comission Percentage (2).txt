#include <iostream>
using namespace std;

int main() {
    double totalSales, percentage = 0;
    cin >> totalSales;

    int category;

    if (totalSales > 1000000)
        category = 1;
    else if (totalSales > 500000)
        category = 2;
    else if (totalSales > 100000)
        category = 3;
    else if (totalSales > 50000)
        category = 4;
    else
        category = 5;

    switch (category) {
        case 1:
            percentage = 0.01;
            break;
        case 2:
            percentage = 0.02;
            break;
        case 3:
            percentage = 0.03;
            break;
        case 4:
            percentage = 0.05;
            break;
        default:
            percentage = 0;
    }

    double commission = percentage * totalSales;
    cout << commission;

    return 0;
}
