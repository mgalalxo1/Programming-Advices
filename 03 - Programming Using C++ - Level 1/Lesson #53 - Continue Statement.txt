#include <iostream>
using namespace std;

int main() {
    int number, sum = 0;

    for (int i = 0; i < 5; i++) {
        cout << "Enter number " << i + 1 << ": ";
        cin >> number;

        if (number >= 50)
            continue; 

        sum += number;
    }

    cout << "Sum: " << sum << endl;

    return 0;
}
