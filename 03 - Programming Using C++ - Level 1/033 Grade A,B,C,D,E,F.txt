#include <iostream>
using namespace std;

int main() {
    int grade;
    cin >> grade;

    switch (grade / 10) {
        case 10:
        case 9:
            cout << "A";
            break;
        case 8:
            cout << "B";
            break;
        case 7:
            cout << "C";
            break;
        case 6:
            cout << "D";
            break;
        case 5:
            cout << "E";
            break;
        default:
            cout << "F";
    }

    return 0;
}
