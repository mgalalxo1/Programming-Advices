#include <iostream>
using namespace std;

int main() {
    int num1, num2, temp;

    cout << "Enter first number: ";
    cin >> num1;

    cout << "Enter second number: ";
    cin >> num2;

    // قبل التبديل
    cout << "\nBefore swapping:\n";
    cout << "num1 = " << num1 << ", num2 = " << num2 << endl;

    // التبديل باستخدام متغير مؤقت
    temp = num1;
    num1 = num2;
    num2 = temp;

    // بعد التبديل
    cout << "\nAfter swapping:\n";
    cout << "num1 = " << num1 << ", num2 = " << num2 << endl;

    return 0;
}
