#include <iostream>
#include <string>
using namespace std;

struct Person {
    string name;
    int age;
    string city;
    string country;
    int monthlySalary;
    int yearlySalary;
    char gender;
    bool married;
};

void printPersonInfo(Person p) {
    cout << "Name    : " << p.name << endl;
    cout << "Age     : " << p.age << " Years" << endl;
    cout << "City    : " << p.city << endl;
    cout << "Country : " << p.country << endl;
    cout << "Monthly Salary : " << p.monthlySalary << endl;
    cout << "Yearly Salary  : " << p.yearlySalary << endl;
    cout << "Gender  : " << p.gender << endl;
    cout << "Married : " << p.married << endl;
}

int main() {
    Person person1;

    person1.name = "Mohammed Abu-Hadhoud";
    person1.age = 44;
    person1.city = "Amman";
    person1.country = "Jordan";
    person1.monthlySalary = 5000;
    person1.yearlySalary = 60000;
    person1.gender = 'M';
    person1.married = true;

    printPersonInfo(person1);

    return 0;
}
