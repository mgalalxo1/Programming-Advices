#include <iostream>
#include <iomanip> 
using namespace std;

int main() {
    const int size = 3;
    int grades[size];
    float sum = 0;

    for (int i = 0; i < size; i++) {
        cout << "Please Enter Grade" << i + 1 << "? ";
        cin >> grades[i];
        sum += grades[i];
    }

    float average = sum / size;

    cout << "************************" << endl;
    cout << fixed << setprecision(4); 
    cout << "The average of grades is " << average << endl;

    return 0;
}
