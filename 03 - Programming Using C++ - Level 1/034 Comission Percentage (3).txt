#include <iostream>
using namespace std;

int main() {
    double totalSales, percentage;
    cin >> totalSales;

    if (totalSales > 1000000)
        percentage = 0.01;
    else if (totalSales > 500000)
        percentage = 0.02;
    else if (totalSales > 100000)
        percentage = 0.03;
    else if (totalSales > 50000)
        percentage = 0.05;
    else
        percentage = 0;

    double commission = totalSales * percentage;
    cout << commission;

    return 0;
}
