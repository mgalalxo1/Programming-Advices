#include <iostream>
using namespace std;

void PrintTaskDurationInSeconds(int days, int hours, int minutes, int seconds) {
    int totalSeconds = (days * 24 * 60 * 60) + (hours * 60 * 60) + (minutes * 60) + seconds;
    cout << totalSeconds << " Seconds";
}

int main() {
    int days, hours, minutes, seconds;
    cin >> days >> hours >> minutes >> seconds;
    PrintTaskDurationInSeconds(days, hours, minutes, seconds);
    return 0;
}
