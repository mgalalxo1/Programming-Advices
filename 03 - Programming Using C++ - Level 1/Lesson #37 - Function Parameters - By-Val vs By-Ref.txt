#include <iostream>
using namespace std;

void swap(int& x, int& y) {
    int temp = x;
    x = y;
    y = temp;
    cout << "After Swap inside function A= " << x << ", B=" << y << endl;
}

int main() {
    int a, b;

    cout << "Please enter the A?\n";
    cin >> a;
    cout << "Please enter the B?\n";
    cin >> b;

    cout << "Before Swap A=" << a << ", B=" << b << endl;

    swap(a, b);

    cout << "After Swap inside Main the function A=" << a << ", B=" << b << endl;

    return 0;
}
