#include <iostream>
using namespace std;

int main() {
    float diameter;
    const float PI = 3.14;

   
    cout << "Enter the diameter of the circle: ";
    cin >> diameter;

   
    float area = (PI * diameter * diameter) / 4;

    
    cout << "The area of the circle is: " << area << endl;

    return 0;
}
